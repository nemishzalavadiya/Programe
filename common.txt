int power(int a, int b)
{
	int res = 1;
	while (b > 0)
	{
		if (b % 2)
		{
			res *= a;
			res %= mod9;
		}
		b /= 2;
		a *= a;
		a %= mod9;
	}
	return res;
}
bool compare(pair<ll, ll> a, pair<ll, ll> b)
{
	if (a.first == b.first)
	{
		return a.second > b.second;
	}
	else
	{
		return a.first < b.first;
	}
}
int sumDigit(ll number)
{
	int sum = 0;
	while (number != 0)
	{
		sum += number % 10;
		number /= 10;
	}
	return sum;
}
